CREATE DATABASE VERIFICATION_CODE;
USE VERIFICATION_CODE;

-- USERS --
CREATE TABLE USERS
(
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(50),
    email VARCHAR(50) UNIQUE,
    password VARCHAR(50)
);

-- VERIFICATION CODES --
CREATE TABLE VERIFICATION_CODES
(
    id INT PRIMARY KEY AUTO_INCREMENT,
    codes BIGINT,
    status INT DEFAULT 1
);
