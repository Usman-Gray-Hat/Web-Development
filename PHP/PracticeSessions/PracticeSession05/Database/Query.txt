-- Creating Database --
CREATE DATABASE USMAN;
USE USMAN;

-- Table For Admins --
CREATE TABLE ADMINS
(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NAME VARCHAR(50) NOT NULL,
    EMAIL VARCHAR(50) NOT NULL,
    PASSWORD VARCHAR(50) NOT NULL
);

-- Inserting Data Into Admins Table --
INSERT INTO ADMINS VALUES(NULL,'Usman','usman@gmail.com','Usman@123');
INSERT INTO ADMINS VALUES(NULL,'Bilal','bilal@gmail.com','Bilal@123');

-- Table For Users --
CREATE TABLE USERS
(
    ID INT PRIMARY KEY AUTO_INCREMENT,
    NAME VARCHAR(50) NOT NULL,
    AGE TINYINT NOT NULL,
    GENDER VARCHAR(50) NOT NULL,
    CITY VARCHAR(50) NOT NULL,
    IMAGE_PATH VARCHAR(400) NOT NULL
);

-- Retrieving Data --
SELECT * FROM ADMINS;
SELECT * FROM USERS;
